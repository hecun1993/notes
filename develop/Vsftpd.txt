vsftpd 完全免费开源的ftp服务器软件

yum install vsftpd -y
rpm -qa | grep vsftpd
/etc/vsftpd/vsftpd.conf

mkdir /home/ftpfile
useradd ftpuser -d /home/ftpfile/ -s /sbin/nologin
chown -R ftpuser.ftpuser  /home/ftpfile/
passwd ftpuser
123456

删除用户：
userdel ftpuser

/etc/vsftpd/ 创建chroot_list，在里面加上ftpuser

编辑vsftpd.conf文件

anonymous_enable=NO

# You may fully customise the login banner string:
ftpd_banner=Welcome to happymmall FTP service.

local_root=/home/ftpfile
use_localtime=yes

chroot_list_enable=YES
# (default follows)
chroot_list_file=/etc/vsftpd/chroot_list

防火墙的配置：编辑/etc/sysconfig/iptables
# vsftpd 
-A INPUT -p TCP --dport 61001:62000 -j ACCEPT
-A OUTPUT -p TCP --sport 61001:62000 -j ACCEPT

-A INPUT -p TCP --dport 20 -j ACCEPT
-A OUTPUT -p TCP --dport 20 -j ACCEPT
-A INPUT -p TCP --dport 21 -j ACCEPT
-A OUTPUT -p TCP --dport 21 -j ACCEPT

在vsftpd.conf中设置防火墙的配置
pasv_min_port=61001
pasv_max_port=62000

如果要看到文件列表，需要设置
setsebool allow_ftpd_full_access=1 


然后在浏览器中访问ftp://192.168.193.100
